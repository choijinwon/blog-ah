let 사용법

TIP

글로벌 변수와, 지역변수에 대한 차이를 볼수 있습니다. for문에서 var 사용했하면 console.log 값이 100으로 값이 output 결과로 나오고 있습니다. var,let 차이는 블럭스코프 단위에서 값을 참조하지 못합니다.

//var참조 가능한영역보여드리는 화면
var global = "글로벌 변수"

function home(){
  var myHome = "지역변수";
  for(var i=0; i < 100; i++){
  }
  console.log(i);
}
home();

//var참조 가능한영역보여드리는 화면
const global = "글로벌 변수"

function home(){
  let myHome = "지역변수";
  for(let i=0; i < 100; i++){
    console.log(i); //정상 실행 
  }
  //추가 로직
   if(true){
     let childif = "childif"
     console.log(childif)
   }
   // console.log(childif) -> Error
  //에러가 발생합니다. let 블럭 스코프단위로 참조를 받고있습니다.
  //console.log(i);
}
home();
